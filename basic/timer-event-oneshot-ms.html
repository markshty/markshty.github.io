ワンショットタイマー


概 要

機　能

ワンショットタイマーは一定時間後に一度だけ呼び出されるタイマーです。

使い方

  • Glib::signal_timeout().connect_once()」において「スロット関数」を指定します。
  • connect_once()関数では、ミリ秒単位でスロット関数の発生間隔を指定します。
  • connect_once()関数を使用する場合はスロット関数の返値は、<void>となります。
  • なお、connect_seconds_once関数では、秒単位でイベントの発生間隔を指定することができます。

サンプル プログラム

update:2025/1/12


#include <gtkmm.h>
#include <ctime>
#include <cmath>
  
class Clock : public Gtk::DrawingArea
{
public:
    Clock();
    virtual ~Clock() = default;
  
protected:
    void   on_draw(const Cairo::RefPtr<Cairo::Context>& cr, int width, int height);
    void   on_timeout();
    double m_radius;
    double m_line_width;
private:
    void   getTime();
    double minutes;
    double hours;
    double seconds;
};
  
Clock::Clock()
 : m_radius( 0.42 ), m_line_width( 0.02 )
{
   // 1000ミリ秒後に、スロット関数Clock::on_timeout()が呼び出される。
   Glib::signal_timeout().connect_once( sigc::mem_fun( *this,  &Clock::on_timeout), 1000 );
    set_draw_func(sigc::mem_fun( *this, &Clock::on_draw ));
}
  
void Clock::on_draw(const Cairo::RefPtr<Cairo::Context>& cr, int width, int height)
{
    double x0, y0, x1, y1, x2, y2;
  
   // 現在時刻の取得 
    getTime();
   // 文字盤
    cr->scale(width, height);
    cr->translate( 0.5, 0.5 );
    cr->set_line_width( m_line_width );
    cr->save();
    cr->set_source_rgba( 0.0, 0.0, 0.0, 1.0 );
    cr->paint();
    cr->restore();
    cr->arc( 0, 0, m_radius, 0, 2 * M_PI );
    cr->save();
    cr->set_source_rgba( 1.0, 1.0, 1.0, 0.8 );
    cr->stroke_preserve();
  
    cr->set_line_cap( Cairo::Context::LineCap::ROUND );
  
   // インデクス(時字)
    for (int i = 0; i < 12; i++)  {
        double inset = 0.08;
        cr->save();
  
        if( i % 3 != 0 ) {
            inset *= 0.85;
            cr->set_line_width( 0.01 );
        }
  
        cr->move_to(
          ( m_radius - inset ) * cos ( i * M_PI / 6 ),
          ( m_radius - inset ) * sin ( i * M_PI / 6 ));
        cr->line_to (
        m_radius * cos ( i * M_PI / 6 ),
        m_radius * sin ( i * M_PI / 6 ));
        cr->stroke();
        cr->restore();
    }
  
   // 秒針
    cr->set_line_width( m_line_width / 3 );
    cr->set_source_rgba( 0.7, 0.7, 0.7, 0.8 );
  
    x1 =  sin( seconds ) * ( m_radius * 0.9 );
    y1 = -cos( seconds ) * ( m_radius * 0.9 );
    cr->move_to( 0, 0 );
    cr->line_to( x1, y1 );
    cr->stroke();
  
   // 分針
    cr->set_source_rgba( 1.0, 1.0, 1.0, 1.0 );
    cr->set_line_width( m_line_width/ 10 );
    cr->set_line_join( Cairo::Context::LineJoin::MITER );
      
    x0 =  sin( minutes + seconds / 60 ) * ( m_radius * 0.8 );
    y0 = -cos( minutes + seconds / 60 ) * ( m_radius * 0.8 );
    x1 =  sin( minutes + seconds / 60 + M_PI / 2 ) * ( m_radius * 0.02 );
    y1 = -cos( minutes + seconds / 60 + M_PI / 2 ) * ( m_radius * 0.02 );
    x2 =  sin( minutes + seconds / 60 - M_PI / 2 ) * ( m_radius * 0.02 );
    y2 = -cos( minutes + seconds / 60 - M_PI / 2 ) * ( m_radius * 0.02 );
  
    cr->move_to( x0, y0 );
    cr->line_to( x1, y1 );
    cr->line_to( x2, y2 );
    cr->close_path();
    cr->fill_preserve();
    cr->stroke();
  
   // 時針
    x0 =  sin( hours + minutes / 12.0 ) * ( m_radius * 0.5 );
    y0 = -cos( hours + minutes / 12.0 ) * ( m_radius * 0.5 );
    x1 =  sin( hours + minutes / 12.0  + M_PI / 2 ) * ( m_radius * 0.02 );
    y1 = -cos( hours + minutes / 12.0  + M_PI / 2 ) * ( m_radius * 0.02 );
    x2 =  sin( hours + minutes / 12.0  - M_PI / 2 ) * ( m_radius * 0.02 );
    y2 = -cos( hours + minutes / 12.0  - M_PI / 2 ) * ( m_radius * 0.02 );
  
    cr->move_to( x0, y0 );
    cr->line_to( x1, y1 );
    cr->line_to( x2, y2 );
    cr->close_path();
    cr->fill_preserve();
    cr->stroke();
}
  
void Clock::getTime()
{
   // 現在時刻の取得
    time_t rawtime;
    time( &rawtime );
    struct tm * timeinfo = localtime( &rawtime) ;
  
   // 時分秒
    hours   = timeinfo->tm_hour * M_PI / 6;
    minutes = timeinfo->tm_min  * M_PI / 30;
    seconds = timeinfo->tm_sec  * M_PI / 30;  
}
  
void Clock::on_timeout()
{
    getTime();
    queue_draw();
   // 次のイベントの呼び出し
    Glib::signal_timeout().connect_once( sigc::mem_fun( *this, &Clock::on_timeout), 1000 );
}
  
class MyWindow : public Gtk::Window
{
public:
    MyWindow();
    virtual ~MyWindow() = default;
protected:
    Clock m_clock;
};
  
MyWindow::MyWindow()
{
    set_title( "Analog Clock" );
    set_size_request( 320, 360 );
    set_child( m_clock );
}
  
int main( int argc, char** argv ) 
{
    auto app = Gtk::Application::create( "gtkmm4.example" );
    return app->make_window_and_run<MyWindow>( argc, argv );
}
