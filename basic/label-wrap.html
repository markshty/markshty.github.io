Labelの中の文字列の折り返し

(update:2026/4/30)


概 要

機　能

labelの中の文字列を折り返します。

使い方

  • set_wrap()を実行します。

サンプル プログラム

リスト


#include <gtkmm.h>

class MyWindow : public Gtk::Window
{
public:
    MyWindow();
    virtual ~MyWindow() = default;
private:
    Gtk::Box m_vbox;
    Gtk::Frame m_frame1, m_frame2;
    Gtk::Label m_label1, m_label2;
};
  
MyWindow::MyWindow()
 : m_vbox( Gtk::Orientation::VERTICAL ),
   m_frame1( "Line Wrapped" ),
   m_frame2( "Fill Line Wrapped" )
{
    set_title("Label( wrapped )");
    m_vbox.set_margin( 15 );
    m_vbox.set_spacing( 10 );
    set_child( m_vbox );
    set_default_size( 200, 350 );
  
  // LineWrapped
    m_label1.set_text( "ABCD EFGH IJKL MNOP QRST UVWX YZ "
                       "ABCD EFGH IJKL MNOP QRST UVWX YZ "
                       "ABCD EFGH IJKL MNOP QRST UVWX YZ "
                       "---- ---- ---- ---- --" );
    m_label1.set_wrap();
    m_frame1.set_child( m_label1 );
    m_vbox.append( m_frame1 ); 
  
   // FilledWrapped
    m_label2.set_text( "ABCD EFGH IJKL MNOP QRST UVWX YZ "
                       "ABCD EFGH IJKL MNOP QRST UVWX YZ "
                       "ABCD EFGH IJKL MNOP QRST UVWX YZ "
                       "---- ---- ---- ---- --" );
    m_label2.set_justify( Gtk::Justification::FILL );
    m_label2.set_wrap();
    m_frame2.set_child( m_label2 );
    m_vbox.append( m_frame2 ); 
}
    
int main(int argc, char* argv[])
{
      auto app = Gtk::Application::create( "gtkmm4.example" );   
      return app->make_window_and_run<MyWindow>(argc, argv);
}

実行結果

サンプルプログラムの上のフレームはset_wrap()のみ

下のフレームは、set_justify()でframeの幅いっぱいに位置揃えを行っています。
