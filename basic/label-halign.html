Labelの配置（水平）

(update:2025/2/1)

概 要

機　能

Window内でのLabelの水平方向の位置を指定します。

使い方

  • set_halign()で左端(Gtk::Align::START)・中央(Gtk::Align::CENTER)・右端(Gtk::Align::END)を指定します。
  • デフォルトは中央(Gtk::Align::CENTER)

サンプル プログラム


#include <gtkmm.h>
  
class MyWindow : public Gtk::Window
{
public:
    MyWindow();
    virtual ~MyWindow() = default;
  
private:
    Gtk::Box m_vbox;
    Gtk::Label m_label1, m_label2, m_label3, m_label4;
};
  
MyWindow::MyWindow()
 : m_vbox( Gtk::Orientation::VERTICAL )
{
    set_title("Label( HAlign )");
    m_vbox.set_margin( 15 );
    m_vbox.set_spacing( 10 );
    set_child( m_vbox );
  
  // START
    m_label1.set_text( "I think,therefore I am" );
    m_label1.set_halign( Gtk::Align::START );
    m_vbox.append( m_label1 ); 
  // CENTER
    m_label2.set_text( "I think,therefore I am" );
    m_label2.set_halign( Gtk::Align::CENTER );
    m_vbox.append( m_label2 );
  // END
    m_label3.set_text( "I think,therefore I am" );
    m_label3.set_halign( Gtk::Align::END );
    m_vbox.append( m_label3 );
  
    m_vbox.set_expand();
} 

int main(int argc, char* argv[])
{
    auto app = Gtk::Application::create("gtkmm4.example");
    return app->make_window_and_run<MyWindow>(argc, argv);
}
