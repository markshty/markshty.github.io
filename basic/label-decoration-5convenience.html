Labelの中の文字列の装飾(Convenience Tags)

(update:2025/2/1)


概 要

機　能

Labelの中の文字列に装飾を加えます（簡略化法）。

使い方

  • gtkmm4では、Pango Markup syntaxを用いてwidgetの中のテキスト(文字列)に装飾を加えることができます。
  • Convenience Tagsを使うとテキスト(文字列)に太字、斜体、下線などを加えることができます。
  • 具体的には、set_markup()の引数として、表示するテキスト(文字列)を指定のタグ(Convenience Tags)で囲みます。
  • タグ(Convenience Tags)で囲まれた範囲の書式が変更されます。
Convenience Tags
<b>～</b> Bold(太字)
<big>～</big> フォントを相対的に大きくする
<i>～</i> Italic(斜体)
<s>～</s> 取り消し線
<sub>～</sub> 添え字
<sup>～</sup> 上付き文字
<small>～</small> フォントを相対的に小さくする
<tt>～</tt> モノスペースフォント
<u>～</u> 下線

サンプル プログラム


#include <gtkmm.h>

class MyWindow : public Gtk::Window
{
public:
    MyWindow();
    virtual ~MyWindow() = default;
private:
    Gtk::Box m_vbox;
    Gtk::Label m_label1, m_label2, m_label3, m_label4, m_label5;
    Gtk::Label m_label6, m_label7, m_label8, m_label9, m_label10;
};
  
MyWindow::MyWindow()
 : m_vbox( Gtk::Orientation::VERTICAL )
{
    set_title("Label(pango markup - Convenience Tags)");
    m_vbox.set_margin( 15 );
    m_vbox.set_spacing( 10 );
    m_vbox.set_expand();
    set_child( m_vbox );
  
   // Bold(太字)
    m_label1.set_markup( "Bold           <b>>The marathon will be held tomorrow</b>" );
    m_label1.set_halign( Gtk::Align::START );
    m_vbox.append( m_label1 ); 
   // big
    m_label2.set_markup( "big            <big>The marathon will be held tomorrow</big>" );
    m_label2.set_halign( Gtk::Align::START );
    m_vbox.append( m_label2 ); 
   // Italic(斜体)
    m_label3.set_markup( "Italic         <i>>The marathon will be held tomorrow</i>" );
    m_label3.set_halign( Gtk::Align::START );
    m_vbox.append( m_label3 );
   // Strikethrough(取り消し線)
    m_label4.set_markup( "Strikethrough  <s>The marathon will be held tomorrow</s>" );
    m_label4.set_halign( Gtk::Align::START );
    m_vbox.append( m_label4 );
   // Subscript(添字)
    m_label5.set_markup( "Subscript      <sub>The marathon will be held tomorrow</sub>" );
    m_label5.set_halign( Gtk::Align::START );
    m_vbox.append( m_label5 ); 
   // Superscript(上付き文字)
    m_label6.set_markup( "Superscript    <sup>The marathon will be held tomorrow</sup>" );
    m_label6.set_halign( Gtk::Align::START );
    m_vbox.append( m_label6 ); 
   // small
    m_label7.set_markup( "smaller        <small>The marathon will be held tomorrow</small>" );
    m_label7.set_halign( Gtk::Align::START);
    m_vbox.append( m_label7 );
   // Monospace Font
    m_label8.set_markup( "Monospace font <tt>The marathon will be held tomorrow</tt>" );
    m_label8.set_halign( Gtk::Align::START );
    m_vbox.append( m_label8 );
   // underline(下線)
    m_label9.set_markup( "underline      <u>The marathon will be held tomorrow</u>" );
    m_label9.set_halign( Gtk::Align::START );
    m_vbox.append( m_label9 );
   // normal
    m_label10.set_markup("normal         The marathon will be held tomorrow" );
    m_label10.set_halign( Gtk::Align::START );
    m_vbox.append( m_label10 );
}
  
int main(int argc, char* argv[])
{
    auto app = Gtk::Application::create( "gtkmm4.example" );   
    return app->make_window_and_run<MyWindow>(argc, argv);
}
