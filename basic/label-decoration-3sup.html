Labelの中の文字列の装飾(上付き文字、添え字)


概 要

  • gtkmm4では、Pango Markup syntaxを用いてwidgetの中のテキスト(文字列)に装飾を加えることができます。
  • The <span> Attributes を使用してフォントのスケールを指定することができます。
  • 具体的には、set_markup()の引数として、表示するテキスト(文字列)を<span>~</span>で囲みます。
  • <span>~</span>で囲まれた範囲の書式が変更されます。
The <span> Attributes
書式 意味 記載例
font_scale 上付き文字 superscript
添え字 subscript
- small-caps

サンプル プログラム

update:2024/11/8


#include <gtkmm.h>

class MyWindow : public Gtk::Window
{
public:
    MyWindow();
    virtual ~MyWindow() = default;
private:
    Gtk::Box m_vbox;
    Gtk::Label m_label1, m_label2, m_label3, m_label4;
};
  
MyWindow::MyWindow()
 : m_vbox( Gtk::Orientation::VERTICAL )
{
    set_title("Label(pango markup - Text Attributes and Markup)");
    m_vbox.set_margin( 15 );
    m_vbox.set_spacing( 10 );
    m_vbox.set_expand();
    set_child( m_vbox );
  
   // superscript
    m_label1.set_text( "X<span font_scale='superscript' rise='5pt'>2</span> + Y<span font_scale='superscript' rise='5pt'>2</span> "
                       "= R<span font_scale='superscript' rise='5pt'>2</span>" );
    m_label1.set_halign( Gtk::Align::CENTER );
    m_label1.set_use_markup( true );
    m_vbox.append( m_label1 );
   // subscript
    m_label2.set_text( "X<span font_scale='subscript'>n+2</span> = X<span font_scale='subscript'>n+1</span>"
                       " + X<span font_scale='subscript'>n</span>" );
    m_label2.set_halign( Gtk::Align::CENTER );
    m_label2.set_use_markup( true );
    m_vbox.append( m_label2 ); 
   // small-caps
    m_label3.set_text( "X<span font_scale='small-caps'>n+2</span> = X<span font_scale='small-caps'>n+1</span>"
                       " + X<span font_scale='small-caps'>n</span>" );
    m_label3.set_halign( Gtk::Align::CENTER );
    m_label3.set_use_markup( true );
    m_vbox.append( m_label3 ); 
}
  
int main(int argc, char* argv[])
{
    auto app = Gtk::Application::create( "gtkmm4.example" );   
    return app->make_window_and_run<MyWindow>(argc, argv);
}
