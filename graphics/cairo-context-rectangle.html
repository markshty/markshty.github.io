長方形(rectangle)

(update:2025/3/8)

概 要

機　能

長方形を描きます。

使い方

rectangle()関数を使用して長方形の左上角の座標と幅、高さを指定します。

void Cairo::Context::rectangle( double x, // 長方形の左上のx座標
double y, // 長方形の左上のy座標
double width, // 長方形の幅
double height ) // 長方形の高さ

サンプル プログラム


#include <gtkmm.h>

class Drawing : public Gtk::DrawingArea {
public:
    Drawing();
    virtual ~Drawing() = default;
protected:
    void on_draw(const Cairo::RefPtr<Cairo::Context>& cr, int width, int height);
};
  
Drawing::Drawing()
{
    set_draw_func( sigc::mem_fun( *this, &Drawing::on_draw));
}
   
void Drawing::on_draw(const Cairo::RefPtr<Cairo::Context>& cr, int width, int height) {          
    cr->rectangle(  50.0,  30.0, 90.0, 90.0 );
    cr->rectangle( 180.0,  30.0, 90.0, 90.0 );
    cr->rectangle( 115.0,  90.0, 90.0, 90.0 );
    cr->rectangle(  50.0, 150.0, 90.0, 90.0 );
    cr->rectangle( 180.0, 150.0, 90.0, 90.0 );
  
    cr->set_line_width( 3.0 );
    cr->set_source_rgb( 0.0, 0.95, 0.0 );
    cr->fill_preserve();
  
    cr->set_source_rgb( 0.0, 0.0, 0.0 );
    cr->stroke();
}
  
class MyWindow : public Gtk::Window
{
public:
    MyWindow();
protected:
    Drawing my_draw;  
};
  
MyWindow::MyWindow()
{
    set_title( "rectangle" );
    set_default_size( 320, 310 );
    set_child( my_draw );
}
  
int main(int argc, char* argv[]) {
    auto app = Gtk::Application::create( "gtkmm4.example" );
    return app->make_window_and_run<MyWindow>( argc, argv );
}
